// Firestore Security Rules for Excelerate App
// Copy this to your Firestore console under Rules tab

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Helper function to check if user is an admin
    function isAdmin() {
      return isAuthenticated() && exists(/databases/$(database)/documents/admins/$(request.auth.uid));
    }
    
    // Helper function to check if document belongs to current user
    function isUserOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    // Users collection - users can only read/write their own documents
    match /users/{userId} {
      allow read: if isAuthenticated() && (isUserOwner(userId) || isAdmin());
      allow create: if isAuthenticated();
      allow update: if isUserOwner(userId) || isAdmin();
      allow delete: if isAdmin();
    }
    
    // Admins collection - allow self-provisioning of admin doc for the authenticated user
    match /admins/{adminId} {
      allow read: if isAdmin() || isUserOwner(adminId);
      allow create: if isAuthenticated() && request.auth.uid == adminId;
      allow update, delete: if false; // Prevent edits/deletes from clients
    }
    
    // Courses collection - read-only to all, users may update enrolledUsers aggregate via client enroll
    match /courses/{courseId} {
      allow read: if true;
      allow create: if isAdmin();
      allow update: if isAdmin() || (isAuthenticated() && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['enrolledUsers']));
      allow delete: if isAdmin();
    }
    
    // Internships collection - read-only to all
    match /internships/{internshipId} {
      allow read: if true;
      allow create: if isAdmin();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // Applications collection - users can read/write their own, admins can read all
    match /applications/{applicationId} {
      allow read: if isAdmin() || (isAuthenticated() && resource.data.userId == request.auth.uid);
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update: if isAdmin();
      allow delete: if isAdmin() || isUserOwner(resource.data.userId);
    }
    
    // Notifications collection - users can only read/write their own notifications
    match /notifications/{notificationId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isUserOwner(resource.data.userId) || isAdmin();
    }
        // Enrollments collection - users can only read/write their own enrollments
        match /enrollments/{enrollmentId} {
          allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
          allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
          allow update: if false; // immutable
          allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
        }
    
    // User Progress collection - users can read/write their own progress
    match /user_progress/{progressId} {
      allow read: if isAuthenticated() && (resource.data.userId == request.auth.uid || isAdmin());
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAdmin();
    }
    
    // Certificates collection - users can read their own, admins can manage all
    match /certificates/{certificateId} {
      allow read: if isAuthenticated() && (resource.data.userId == request.auth.uid || isAdmin());
      allow create: if isAdmin(); // Only issued by system/admin
      allow update, delete: if isAdmin();
    }
    
    // Support Tickets collection - users can read/write their own, admins can read all
    match /support_tickets/{ticketId} {
      allow read: if isAuthenticated() && (resource.data.userId == request.auth.uid || isAdmin());
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update: if isAdmin() || (isAuthenticated() && resource.data.userId == request.auth.uid);
      allow delete: if isAdmin();
    }
    
    // Feedback collection - users can submit, admins can read all
    match /feedback/{feedbackId} {
      allow read: if isAdmin() || (isAuthenticated() && resource.data.userId == request.auth.uid);
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAdmin();
    }
  }
}
