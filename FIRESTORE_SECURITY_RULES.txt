// Firestore Security Rules for Excelerate App
// Copy this to your Firestore console under Rules tab

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Helper function to check if user is an admin
    function isAdmin() {
      return isAuthenticated() && exists(/databases/$(database)/documents/admins/$(request.auth.uid));
    }
    
    // Helper function to check if document belongs to current user
    function isUserOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    // Users collection - users can only read/write their own documents
    match /users/{userId} {
      allow read: if isAuthenticated() && (isUserOwner(userId) || isAdmin());
      allow write: if isUserOwner(userId);
      allow create: if isAuthenticated();
      allow delete: if isAdmin();
    }
    
    // Admins collection - only admins can read, only system can write
    match /admins/{adminId} {
      allow read: if isAdmin() || isUserOwner(adminId);
      allow write: if false; // Admins are created via Firebase Console or custom claims
    }
    
    // Courses collection - authenticated users can read all, only admins can write
    match /courses/{courseId} {
      allow read: if isAuthenticated();
      allow write: if isAdmin();
      allow create: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // Internships collection - authenticated users can read all, only admins can write
    match /internships/{internshipId} {
      allow read: if isAuthenticated();
      allow write: if isAdmin();
      allow create: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // Applications collection - users can read/write their own, admins can read all
    match /applications/{applicationId} {
      allow read: if isAdmin() || (isAuthenticated() && resource.data.userId == request.auth.uid);
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update: if isAdmin();
      allow delete: if isAdmin() || isUserOwner(resource.data.userId);
    }
    
    // Notifications collection - users can only read their own notifications
    match /notifications/{notificationId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow write: if false; // Notifications are created by backend/Cloud Functions
      allow delete: if isUserOwner(resource.data.userId) || isAdmin();
    }
  }
}
